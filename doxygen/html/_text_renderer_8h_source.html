47  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);

92  void renderText(std::string text, GLfloat x, GLfloat y, GLfloat scale, glm::vec3 color = glm::vec3(1.0f)) {
93  this->shader.Use();
94  this->shader.SetVector3f("textColor", color);
95  glActiveTexture(GL_TEXTURE0);
96  glBindVertexArray(this->VAO);
97 
98  // Iterate through all characters
99  std::string::const_iterator c;
100  for (c = text.begin(); c != text.end(); c++) {
101  Character ch = characters[*c];
102 
103  GLfloat xpos = x + ch.bearing.x * scale;
104  GLfloat ypos = y + (this->characters['H'].bearing.y - ch.bearing.y) * scale;
105 
106  GLfloat w = ch.size.x * scale;
107  GLfloat h = ch.size.y * scale;
108  // Update VBO for each character
109  GLfloat vertices[6][4] = {
110  {xpos, ypos + h, 0.0, 1.0},
111  {xpos + w, ypos, 1.0, 0.0},
112  {xpos, ypos, 0.0, 0.0},
113 
114  {xpos, ypos + h, 0.0, 1.0},
115  {xpos + w, ypos + h, 1.0, 1.0},
116  {xpos + w, ypos, 1.0, 0.0}
117  };
118  // Render glyph texture over quad
119  glBindTexture(GL_TEXTURE_2D, ch.textureID);
120  // Update content of VBO memory
121  glBindBuffer(GL_ARRAY_BUFFER, this->VBO);
122  glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(vertices),
123  vertices); // Be sure to use glBufferSubData and not glBufferData
124 
125  glBindBuffer(GL_ARRAY_BUFFER, 0);
126  // Render quad
127  glDrawArrays(GL_TRIANGLES, 0, 6);
128  // Now advance cursors for next glyph
129  x += (ch.advance >> 6) * scale; // Bitshift by 6 to get value in pixels (1/64th times 2^6 = 64)
130  }
131  glBindVertexArray(0);
132  glBindTexture(GL_TEXTURE_2D, 0);
133  }
