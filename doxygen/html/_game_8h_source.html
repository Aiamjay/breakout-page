219  void doCollisions() {
220  // collision with bricks
221  for (auto &brick : this->levels[this->currentLevel].bricks) {
222  if (!brick.isDestroyed) {
223  Collision result = checkCollision(*ball, brick);
224  if (std::get<0>(result) == GL_TRUE) {
225  if (!brick.isSolid) {
226  SoundEngine->play2D("../media/music/bleep.wav", GL_FALSE);
227  brick.isDestroyed = GL_TRUE;
228  score += 1;
229  } else {
230  shakeTime = 0.05f;
231  effects->shake = true;
232  SoundEngine->play2D("../media/music/solid.wav", GL_FALSE);
233  }
234  Direction dir = std::get<1>(result);
235  glm::vec2 difference = std::get<2>(result);
236  if (dir == LEFT || dir == RIGHT) {
237  ball->velocity.x = -ball->velocity.x;
238  GLfloat penetration = ball->radius - std::abs(difference.x);
239  if (dir == LEFT) ball->position.x += penetration;
240  else ball->position.x -= penetration;
241  } else if (dir == DOWN || dir == UP) {
242  ball->velocity.y = -ball->velocity.y; // 反转垂直速度
243  GLfloat penetration = ball->radius - std::abs(difference.y);
244  if (dir == UP) ball->position.y -= penetration; // 将球上移
245  else ball->position.y += penetration; // 将球下移
246  }
247  }
248  }
249  }
250 
251  // collision with paddle
252  if (ball->stuck == GL_FALSE) {
253  Collision result = checkCollision(*ball, *player);
254  if (std::get<0>(result) == GL_TRUE) {
255  GLfloat centerPaddle = player->position.x + player->size.x / 2.0f;
256  GLfloat distance = (ball->position.x + ball->radius) - centerPaddle;
257  GLfloat percentage = distance / (player->size.x / 2);
258  GLfloat strength = 2.0f;
259  glm::vec2 oldVelocity = ball->velocity;
260  ball->velocity.x = INITIAL_BALL_VELOCITY.x * percentage * strength;
261  ball->velocity.y = -1 * abs(ball->velocity.y);
262  ball->velocity = glm::normalize(ball->velocity) * glm::length(oldVelocity);
263  SoundEngine->play2D("../media/music/paddle.wav", GL_FALSE);
264  }
265  }
266  }

344  static Collision checkCollision(BallObject &ball, GameObject &brick) {
345  glm::vec2 ballCenter(ball.position + ball.radius);
346  glm::vec2 halfWidthHeight(brick.size.x / 2.f, brick.size.y / 2.f);
347  glm::vec2 brickCenter(brick.position + halfWidthHeight);
348  glm::vec2 difference = ballCenter - brickCenter;
349  glm::vec2 clampedDiff = glm::clamp(difference, -halfWidthHeight, halfWidthHeight);
350  glm::vec2 closest = brickCenter + clampedDiff;
351  difference = closest - ballCenter;
352  if (glm::length(difference) < ball.radius) {
353  return std::make_tuple(GL_TRUE, vectorDirection(difference), difference);
354  } else {
355  return std::make_tuple(GL_FALSE, UP, glm::vec2(0.0f));
356  }
357  }

359  static Direction vectorDirection(glm::vec2 difference) {
360  GLfloat max = 0.f;
361  GLuint bestMatch = -1;
362  for (unsigned int i = 0; i < 4; i++) {
363  GLfloat dotProduct = glm::dot(glm::normalize(difference), DIRECTIONS.at(i));
364  if (dotProduct > max) {
365  max = dotProduct;
366  bestMatch = i;
367  }
368  }
369  return (Direction) bestMatch;
370  }
